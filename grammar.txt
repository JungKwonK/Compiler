ROOT		-> 	PACKAGE
			| 	PACKAGE IMPORT_REC
PACKAGE 	-> 	package PKG_ID
PKG_ID	->	^[a-zA-Z0-9]+$
IMPORT_REC 	->	IMPORT IMPORT_REC
			| 	IMPORT
IMPORT 		-> 	import IMP_ID STR_LIT
			| 	import IMP_PATH
			|	import ( IMP_REC )
IMP_REC 	-> 	IMP_ID STR_LIT IMP_REC
			| 	IMP_ID STR_LIT
			| 	IMP_PATH
IMP_ID 		-> 	^([._]|[a-zA-Z0-9]+)$
STR_LIT	-> 	^".*"$
