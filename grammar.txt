ROOT			-> 	PACKAGE
				| 	PKG_STMT IMP_STMT_REC
PKG_STMT 		-> 	package PKG_ID
PKG_ID			->	^[a-zA-Z0-9]+$
IMP_STMT_REC 	->	IMP_STMT IMP_STMT_REC
				| 	IMP_STMT
IMP_STMT 		-> 	import IMP_ID STR_LIT
				| 	import IMP_PATH
				|	import ( IMP_REC )
IMP_REC 		-> 	IMP_ID STR_LIT IMP_REC
				| 	IMP_ID STR_LIT
				| 	IMP_PATH
IMP_ID 			-> 	^([._]|[a-zA-Z0-9]+)$
STR_LIT			-> 	^".*"$
